/**
 * Light session wrapper
 */
class Session {
    
    public static function getSessionId(){
        return session_id();
    }
    
    public static function get($name){
        return $_SESSION[$name];
    }
    
    public static function set($name, $value){
        $_SESSION[$name] = $value;
    }
    
    

}
